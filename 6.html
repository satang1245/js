함수

일련의 과정을 문으로 구현하고 코드 블록으로 감싸서 하나의 실행 단위로 정의한 것

// 함수정의
// 매개변수 or 인자. paramater, 인수 : argument 
function 함수명(매개변수, 매개변수) {
    return 반환값;
}

// 함수호출
함수명(인수, 인수);
함수는 왜 사용하는가?

1. 코드의 재사용.
2.유지보수의 편의성
3.코드의 신뢰성
4.코드의 가독성

함수 정의 방식 예시
함수 선언문 
function sum(x,y) {
  return x+y;
}
함수 표현식 
var sum = function (x,y) {
  return x+y;
}
생성자 함수 
var sum = new function('x','y', 'return x + y');
화살표 함수(ES6) 
var sum = (x,y) => {
    return x + y;
}
// 축약 - return 값이 한 줄일 경우 중괄호{} 생략가능.
var add = (x,y) => x + y;

// 축약 - 인수가 하나일 경우 괄호() 생략가능.
var add = x => x + 1;

*피연산자를 이용하여 함수 선언시 함수 표현식으로 해석된다.

//피연사자
( function bar() {
    console.log('bar');

    let result = '청년';
    if ( age > 19 ) {
        result = '어른';

    return result;
}

function adultGame(age, checkAge) {
    if ( !checkAge ) {
        checkAge = 19;
    }
    if ( age < checkAge ) {
        console.log('넌 어른이 아니야!');
        return;
    }
    console.log('난 어른이지롱!! 😬');  
}

    return 'Hello!~ My name is

function getName(firstName, lastName) {

function sum() {
    console.log( arguments );

function sum() {
    let total = 0;
    for (let i=0; i<arguments.length; i++) {
        total += arguments[i];
    }
    return total;
}

function sum(a) {
    arguments[0] = 100;
    return a;
}
