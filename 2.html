변수 선언 및 할당, 그리고 연산자

변수 선언 및 할당

var str;  // 변수 선언
str = 'Hello World';  // 변수 할당
str = 'Nice to meet you!'; // 변수 재할당

// 변수 선언 및 할당
var person = 'Son',
    address = 'Seoul',
    age = 29;

// 중복선언
var x = 10; // 10;
console.log(x) // 10
var x = 120; // 120;
console.log(x) // 120

명시적 타입변환, 암묵적 타입전환

모든 값에는 타입이 있다. 개발자의 의도적으로 타입을 변환할 수 도 있는데,
이를 명시적 타입 변환 또는 타입 캐스팅 이라 한다.
그 외에 자바스크립트 엔진에 의해 암묵적으로 자동 변환될 수 있는데,
이를 암묵적 타입변환 또는 타입 강제 변환 이라고 한다.

var x = 10;

// 명시적 타입 변환
var str = x.toString(); // 숫자를 문자열로 타입 캐스팅한다.
console.log(typeof str); // string

// 암묵적 타입 변환
// 숫자 타입 x의 값을 바탕으로 새로운 문자열 타입의 값을 생성해 표현식을 평가한다.
var str = x + '';
console.log(typeof str); // string

스코프(scope)

식별자(변수 및 함수 등)를 사용할 수 있는 유효범위
변수는 전역 또는 코드 블록(if, for, while, try/catch 등)이나 함수내에 선언하여 사용되는데,
이 때 변수는 중첩되어 선언이 될 수 있으며, 다른 코드가 자신을 참조할 수 있는 유효한 범위를
가지는데 이를 스코프라 한다.

  • 전역 스코프(Global scope)
    코드 어디에서든 참조할 수 있다.
  • 지역 스코프(Local scope or Function-lavel scope)
    자신을 포함한 함수 및 하위 함수에서만 참조할 수 있다.
var global = 'global'; // 전역 스코프

function foo() {
  var local = 'local'; // 지역 스코프
  console.log(global);
  console.log(local);
}
foo();

console.log(global);
console.log(local); // Uncaught ReferenceError: local is not defined

---

var x = 10;

function foo(){
  var x = 100;
  console.log('A:', x);

  function bar(){   // 내부함수
    x = 1000;
    console.log('B:', x); // ?
  }

  bar();
}
foo();
console.log(x); 

---

var x = 1;

function foo() {
  var x = 10;
  bar();
}

function bar() {
  console.log(x);
}

foo(); // ?
bar(); // ?

---

var x = 10; // 전역 변수

function foo () {
  // 선언하지 않은 변수
  y = 20;
  console.log(x + y);
}

foo(); // 30

console.log(window.x); // 10
console.log(window.y); // 20

delete x; // 전역 변수는 삭제되지 않는다.
delete y; // 프로퍼티는 삭제된다.

console.log(window.x); // 10
console.log(window.y); // undefined
