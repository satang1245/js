조건문, 반복문

조건문

주어진 조건에 따른 평가 결과에 따른 코드 블럭문을 실행한다.

# if 문

// if문
if ( 조건식1 ) { 
    // 조건식1이 참이면 코드 블럭 실행
} else if ( 조건식2 ) {
    // 조건식2가 참이면 코드 블럭 실행
} else if ( 조건식3 ) {
    // 조건식3가 참이면 코드 블럭 실행
} else {
    // 모든 조건식에 맞지 않다면..코드 블럭 실행
}
*코드 블럭내에 문이 하나뿐이라면 중괄호({})를 생략할 수 있다.
소스보기 
var num = 2;
var kind = null;

if (num > 0)      kind = '양수';
else if (num < 0) kind = '음수';
else              kind = '영';
                        
console.log(kind);
*삼항연산자를 이용하여 축약식으로 표현도 가능하다.
소스보기 
// 삼항연산자 표기법 - 조건식에 맞춰 true 또는 false의 값을 반환해 줌.
// 조건식 ? ture경우의 값 : false경우의 값;

var num = 2;
var kind = (num > 0) ? '양수' : ((num < 0) ? '음수' : '영');
var kind = num ? ((num > 0) ? '양수' : '음수') : '영'; // 숫자에서 0은 false;


console.log(kind);

# switch 문

// switch문
switch (표현식) {
    case 표현식1:
        switch 문의 표현식과 표현식1이 일치하면 실행될 문;
        break;
    case 표현식2:
        switch 문의 표현식과 표현식2가 일치하면 실행될 문;
        break;
    default:
        switch 문의 표현식과 일치하는 표현식을 갖는 case 문이 없을 때 실행될 문;
}

// 예시
var month = 3;
var monthName = '';
switch (month) {
    case 1 :
        monthName = 'January';
        break;
    case 2 :
        monthName = 'February';
        break;
    case 3 :
        monthName = March;
        break;
    ...
    default :
        monthName = 'Invalid month';
        break;
}
*switch문은 여러개를 묶어서 표현이 가능하며 default문은 생략이 가능하다
소스보기 
var year = 2000; // 2000년은 윤년으로 2월이 29일이다.
var month = 2;
var days = 0;

switch (month) {
    case 1: case 3: case 5: 
    case 7: case 8: case 10: case 12:
        days = 31;
        break;
    case 4: case 6: case 9: case 11:
        days = 30;
        break;
    case 2:
        days = ((year % 4 === 0 && year % 100 !== 0) || (year % 400 === 0)) ? 29 : 28; // 윤년 계산 알고리즘
        break;
}

console.log(days); // 29
*배열을 이용하여 축약식으로 표현할 수 있다.
소스보기 
var month = 3;
var monthNameArr = ['January','February','March','April','May','June','July','August','September','October','November','December'];
var monthName = monthNameArr[month - 1]; // 배열의 접근의 경우 배열의 index는 0번부터 시작함.

반목문

조건식이 거짓이 될 때까지 코드블럭을 실행한다.

# for 문

for 문은 조건식이 거짓으로 판별될 때까지 코드 블록을 반복 실행

for (초기화식; 조건식; 증감식) {
    조건식이 참인 경우 반복 실행될 문;
}
예시보기 
for (var i = 1; i < 3; i++) {
    console.log(i);
}

// 두 개의 주사위를 던졌을 때, 
// 두 눈의 합이 6이 되는 모든 경우의 수를 출력하는 예제.
for (var i = 1; i <= 6; i++) {
    for (var j = 1; j <= 6; j++) {
        if (i + j === 6) console.log(`[${i}, ${j}]`);
    }
}

# while 문

while 문은 조건식이 거짓으로 판별될 때까지 코드 블록을 반복 실행

while (조건식) {
    조건식이 참인 경우 반복 실행될 문;
}
예시보기 
var count = 1; // 초기화식;
while (count < 3) { // 조건식
    console.log(count);
    count += 1; // 증감식
}

# do...while 문

do...while 문은 코드 블럭을 실행하고 조건식을 평가.*따라서 코드 블럭은 무조건 한번 이상 실행

do {
    조건식이 참인 경우 반복 실행될 문;
} while (조건식)
예시보기 
var count = 10; // 초기화식;
do { // 조건식
    console.log('count: ', count);
    count += 1; // 증감식
} while (count < 5)

# for...in & for...of

for...in 문
  • 객체의 모든 열거가능한 속성에 대해 반복 실행
  • 객체의 key값에 접근 가능, value값에는 직접 접근 불가
  • 모든 객체에서 사용 가능
// for in문
for (const variable in object) {
    statement;
}

// 예제
const obj = { a: 1, b: 2, c: 3 };

for (const prop in obj) {
  console.log(`${prop}: ${obj[prop]}`);
}
//output: a: 1, b: 2, c: 3 

const obj2 = ['a','b','c'];
for (const prop in obj2) {
    console.log(`${prop}: ${obj2[prop]}`);
}
//output: 1: a, 2: b, 3: b
for...of 문(ES6에 추가)
  • 반복 가능한 객체(iterable)를 순회할 수 있도록 해줌
  • Array, Map, Set, arguments 등이 해당됨 (Object는 해당 X)
  • 컬렉션 전용으로 객체보다는, [Symbol.iterator] 속성이 있는 모든 컬렉션 요소에 사용
// Array
let iterable = [10, 20, 30];
for (let value of iterable) {
    console.log(value);
}
// output: 10,20,30

---
// String
let iterable = 'Hello';
for (let value of Hello) {
    console.log(value);
}
// output: "h","e","l","l","o";

---
// Object
let obj = { a: 1, b: 2, c: 3 };
for (const prop in obj) {
  console.log(`${prop}: ${obj[prop]}`);
}
// output: Uncaught TypeError: obj is not iterable

---
// DOM 컬렉션에 대한 반복
let articleParagraphs = document.querySelectorAll("p");

for (let paragraph of articleParagraphs) {
  console.log( paragraph );
  // paragraph.classList.add('read');
}
// output: NodeList(9) [p, p, p, p, p, p, p, p, p]

for...in문과 for...of의 차이점

for in: 객체 순환(Key값을 반환)
for of: 배열 순환(value값을 반환)

# break & continue

break문
반복문(for, for-in, for-of, while, do-while)의 코드 블록 실행 중 현 지점에서 중단하고 반복문 자체를 벗어날 때 사용함.

// break문
var count = 0;
for(var i=0; i<20; i++) {
    if ( i > 5 ) {
        break;
    }
    count++;
}
console.log(count);

continue문
반복문(for, for-in, for-of, while, do-while)의 코드 블록 실행을 현 지점에서 중단하고 다음 증감식으로 이동함. 

// continue문을 이용한 홀수만 출력하기
for(var i=0; i<10; i++) {
    if ( i % 2 == 0 ) {
        continue;
    }
    console.log('result i값 :', i);
}
